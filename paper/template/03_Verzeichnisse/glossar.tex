\newglossaryentry{GPS}{
	name={GPS},
	description={Das Global Positioning System (GPS) ist ein von den Vereinigten Staaten von Amerika
	entwickeltes globales Navigationssystem, welches im zivilen, kommerziellen, wissenschaftlichen und
	militärischen Bereich Anwendung findet, um die Position von GPS-Empfängern zu bestimmen.
	(vgl. \cite{GPS2008})
	}
}

\newglossaryentry{Overhead}{
	name={Overhead},
	description={Als Overhead werden Metadaten bezeichnet, welche zusätzlich zu den eigentlichen
	Nutzdaten anfallen. Beispiele für Overhead sind Prüfsummen, Netzwerkadressen oder
	Codierungszeichen.
	(vgl. \cite{Cavanaugh})
	}
}

\newglossaryentry{Mesh-Netzwerk}{
	name={Mesh-Netzwerk},
	description={In Mesh-Netzwerken verbinden sich die Teilnehmer direkt miteinander und kommunizieren
	direkt miteinander. (vgl. \cite{Cilfone2019}) Dies steht im Kontrast zur
	Client-Server-Kommunikation, wie sie beispielsweise bei WLAN genutzt wird. Hierbei verbinden
	sich alle Netzteilnehmer mit einer zentralen Vermittlungsstelle und jegliche
	Kommunikation wird über diese weitergeleitet.
	}
 }
